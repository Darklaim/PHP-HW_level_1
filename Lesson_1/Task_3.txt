$a = 5;
$b = '05';
var_dump($a == $b); // Почему true? - Нестрогое равенство, сравнение значений, в данном случае 5 = 5 (Значение 0 перед 5 не имеет существенного значения);
var_dump((int)'012345'); // Почему 12345?  Значение приводится к целому числу, в данном случае 0 опускается.
var_dump((float)123.0 === (int)123.0); // Почему false? Строгое равенство, разные типы float и int.
var_dump((int)0 === (int)'hello, world'); // Почему true? Строгое равенство, во втором значении сравнения нет цифр, т.о. значение априори устанавливается, как "0".
